-- Crea procedimiento para escritura del archivo .txt al insertar registros en la tabla.
USE GESTITRV;
GO

CREATE PROCEDURE WriteToFile @TextToWrite NVARCHAR(MAX)
AS BEGIN
  DECLARE @FilePath NVARCHAR(255)
  DECLARE @cmd SYSNAME
  SET @FilePath = 'C:\DATOS\PDAnewReg.txt'
  --SET @cmd = 'echo ' + @TextToWrite + ' > ' + @FilePath;
  SET @cmd = 'echo ' + @TextToWrite + ' > C:\PDAnewReg.txt';
  EXEC xp_cmdshell @cmd;
END
GO

-- Disparador al INSERTAR datos en la tabla.
-- Quremos escribir en un archivo.txt
CREATE TRIGGER [dbo].[trgAfterInsertOnPeticion]
ON [PETICION]
AFTER INSERT
AS
BEGIN
    DECLARE @NewRecord NVARCHAR(MAX)

    SELECT @NewRecord = 
        'Nuevo registro insertado: ' +
        'PDA=' + CAST([PDA] AS NVARCHAR) + ', ' +
        'TIPO=' + CAST([TIPO] AS NVARCHAR) + ', ' +  -- Asegúrate de incluir todos los campos necesarios
        'ESTADO=' + CAST([ESTADO] AS NVARCHAR) + ', ' +
        'VALOR=' + [VALOR] + ', ' +
        'CAMPO01=' + [CAMPO01] + ', ' +
        'CAMPO02=' + [CAMPO02] + ', ' +
        'CAMPO03=' + [CAMPO03] + ', ' +
        'CAMPO04=' + [CAMPO04] + ', ' +
        'CAMPO05=' + [CAMPO05]
    FROM INSERTED
    EXEC [dbo].[WriteToFile] @NewRecord
END

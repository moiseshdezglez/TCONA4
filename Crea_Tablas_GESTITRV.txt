CREATE VIEW [dbo].[TeleoperadoraVisita]
AS
SELECT   dbo.OPVIS.FECHA, dbo.OPVIS.CODIGO, dbo.OPVIS.NOMBRE, dbo.OPVIS.APELLIDO1, dbo.OPVIS.APELLIDO2, dbo.Ven.nombre AS Teleoperadora, dbo.OPVIS.OBS, 
                         dbo.OPCON.Fecha AS FechaContrato, dbo.OPCON.Numero
FROM         dbo.Ven RIGHT OUTER JOIN
                         dbo.OPVIS ON dbo.Ven.codigo = dbo.OPVIS.TELEOPERADORA LEFT OUTER JOIN
                         dbo.OPCON ON dbo.OPVIS.CODIGO = dbo.OPCON.Visita
WHERE     (dbo.OPVIS.FECHA BETWEEN N'12/01/2023' AND N'12/01/2023')
GO
